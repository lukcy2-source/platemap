    // --- 1. Firebase 配置 (已為你替換完成) ---

        databaseURL: "https://geomap-1d09b-default-rtdb.asia-southeast1.firebasedatabase.app",
        projectId: "geomap-1d09b",
        storageBucket: "geomap-1d09b.firebasestorage.app",
        messagingSenderId: "618061351875",
        appId: "1:618061351875:web:cf2ee7e4a19ccd3b7a0b76",
        measurementId: "G-WWCPHE2JJW"
    };
    
    // 初始化 Firebase
    firebase.initializeApp(firebaseConfig);
    const db = firebase.database().ref('global_comments');

    // --- 2. 初始化地圖 ---
    var map = L.map('map').setView([20, 0], 2);

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
        attribution: '© OpenStreetMap contributors'
    }).addTo(map);

    // 載入全球板塊邊界 GeoJSON
    fetch('https://raw.githubusercontent.com/fraxen/tectonicplates/master/GeoJSON/PB2002_boundaries.json')
        .then(res => res.json())
        .then(data => {
            L.geoJSON(data, {
                style: { color: "#ff4500", weight: 3, opacity: 0.7 }
            }).addTo(map);
        });

    // --- 3. 實時讀取：當資料庫有新留言時，自動在地圖上生成標記 ---
    db.on('child_added', (snapshot) => {
        const data = snapshot.val();
        const marker = L.marker([data.lat, data.lng]).addTo(map);
        
        const content = `
            <div class="custom-popup">
                <strong>👤 ${data.user}</strong><br>
                <b>地貌成因：</b><br>${data.text}<br>
                <small style="color:#bdc3c7;">📅 ${data.time}</small>
            </div>
        `;
        marker.bindPopup(content);
    });

    // --- 4. 點擊功能：新增留言並存入雲端 ---
    map.on('click', function(e) {
        var user = prompt("請輸入你的名字：");
        if (!user) return;

        var text = prompt("請描述此處的地貌形成過程：");
        if (!text) return;

        const lat = e.latlng.lat;
        const lng = e.latlng.lng;

        // 將資料推送到 Firebase
        db.push({
            user: user,
            text: text,
            lat: lat,
            lng: lng,
            time: new Date().toLocaleString()
        });
    });
</script>
